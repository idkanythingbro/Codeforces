//Java
import java.util.*;
public class Main{
	static Scanner sc = new Scanner(System.in);
	public static void main(String args[]){
		int n= sc.nextInt();
		int ans = 0;
		while(n > 0){
			int p = sc.nextInt();
			int v = sc.nextInt();
			int t = sc.nextInt();
			if((p + v + t) >= 2){
				ans++;
			}
			n--;
		} 
		System.out.println(ans);
	}
}
